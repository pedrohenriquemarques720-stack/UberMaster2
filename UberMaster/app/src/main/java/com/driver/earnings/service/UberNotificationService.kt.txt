package com.driver.earnings.service

import android.service.notification.NotificationListenerService
import android.service.notification.StatusBarNotification
import android.content.SharedPreferences
import android.content.Intent
import android.content.Context
import androidx.preference.PreferenceManager
import android.util.Log
import java.util.regex.Pattern

class UberNotificationService : NotificationListenerService() {
    
    private lateinit var sharedPreferences: SharedPreferences
    private val TAG = "UberNotification"
    
    override fun onCreate() {
        super.onCreate()
        sharedPreferences = PreferenceManager.getDefaultSharedPreferences(this)
        Log.d(TAG, "Service de notificação criado")
    }
    
    override fun onNotificationPosted(sbn: StatusBarNotification) {
        // Verifica se é notificação da Uber
        if (sbn.packageName == "com.ubercab.driver") {
            val notification = sbn.notification
            val extras = notification.extras
            
            // Extrai o texto da notificação
            val title = extras.getCharSequence(android.app.Notification.EXTRA_TITLE).toString()
            val text = extras.getCharSequence(android.app.Notification.EXTRA_TEXT).toString()
            val fullText = "$title $text"
            
            Log.d(TAG, "Notificação recebida: $fullText")
            
            // Extrai valor e distância usando regex
            val amount = extractAmount(fullText)
            val distance = extractDistance(fullText)
            
            if (amount != null && distance != null) {
                saveEarningData(amount, distance)
                // CORREÇÃO 1: Dispara intent para iniciar/atualizar o OverlayService
                startOverlayService()
            } else {
                Log.d(TAG, "Não foi possível extrair valor ou distância da notificação")
            }
        }
    }
    
    private fun extractAmount(text: String): Double? {
        // Regex para encontrar padrões como R$ 15,50 ou R$15,50 ou R$ 15.50
        val amountPattern = Pattern.compile("R\\$\\s*(\\d+[.,]\\d{2})")
        val matcher = amountPattern.matcher(text)
        
        return if (matcher.find()) {
            val amountStr = matcher.group(1).replace(",", ".")
            try {
                amountStr.toDouble()
            } catch (e: NumberFormatException) {
                Log.e(TAG, "Erro ao converter valor: $amountStr")
                null
            }
        } else {
            // Tenta outro padrão: apenas números com símbolo R$
            val alternativePattern = Pattern.compile("(\\d+[.,]\\d{2})")
            val altMatcher = alternativePattern.matcher(text)
            if (altMatcher.find()) {
                val amountStr = altMatcher.group(1).replace(",", ".")
                try {
                    amountStr.toDouble()
                } catch (e: NumberFormatException) {
                    null
                }
            } else {
                null
            }
        }
    }
    
    private fun extractDistance(text: String): Double? {
        // Regex para encontrar padrões como 8.4 km, 8,4 km, 8.4km, 8,4km
        val distancePattern = Pattern.compile("(\\d+[.,]\\d+)\\s*km", Pattern.CASE_INSENSITIVE)
        val matcher = distancePattern.matcher(text)
        
        return if (matcher.find()) {
            val distanceStr = matcher.group(1).replace(",", ".")
            try {
                distanceStr.toDouble()
            } catch (e: NumberFormatException) {
                Log.e(TAG, "Erro ao converter distância: $distanceStr")
                null
            }
        } else {
            // Tenta encontrar números que parecem distância (ex: 8.4 sozinho)
            val alternativePattern = Pattern.compile("(\\d+[.,]\\d+)")
            val altMatcher = alternativePattern.matcher(text)
            if (altMatcher.find()) {
                val distanceStr = altMatcher.group(1).replace(",", ".")
                try {
                    distanceStr.toDouble()
                } catch (e: NumberFormatException) {
                    null
                }
            } else {
                null
            }
        }
    }
    
    private fun saveEarningData(amount: Double, distance: Double) {
        val editor = sharedPreferences.edit()
        editor.putFloat("last_amount", amount.toFloat())
        editor.putFloat("last_distance", distance.toFloat())
        editor.apply()
        
        Log.d(TAG, "Dados salvos - Valor: R$ $amount, Distância: $distance km")
    }
    
    // CORREÇÃO 1: Método para iniciar o OverlayService
    private fun startOverlayService() {
        try {
            val intent = Intent(this, OverlayService::class.java)
            if (android.os.Build.VERSION.SDK_INT >= android.os.Build.VERSION_CODES.O) {
                startForegroundService(intent)
            } else {
                startService(intent)
            }
            Log.d(TAG, "OverlayService iniciado/atualizado")
        } catch (e: Exception) {
            Log.e(TAG, "Erro ao iniciar OverlayService: ${e.message}")
        }
    }
    
    override fun onNotificationRemoved(sbn: StatusBarNotification) {
        // Opcional: lidar com notificações removidas
        Log.d(TAG, "Notificação removida: ${sbn.packageName}")
    }
    
    override fun onDestroy() {
        super.onDestroy()
        Log.d(TAG, "Service de notificação destruído")
    }
    
    companion object {
        fun saveCarSettings(context: Context, consumption: Float, fuelPrice: Float) {
            val prefs = PreferenceManager.getDefaultSharedPreferences(context)
            val editor = prefs.edit()
            editor.putFloat("car_consumption", consumption)
            editor.putFloat("fuel_price", fuelPrice)
            editor.apply()
            Log.d("UberNotification", "Configurações do carro salvas - Consumo: $consumption km/L, Preço: R$ $fuelPrice")
        }
        
        fun getCarSettings(context: Context): Pair<Float, Float> {
            val prefs = PreferenceManager.getDefaultSharedPreferences(context)
            val consumption = prefs.getFloat("car_consumption", 10f)
            val fuelPrice = prefs.getFloat("fuel_price", 5.50f)
            return Pair(consumption, fuelPrice)
        }
    }
}
